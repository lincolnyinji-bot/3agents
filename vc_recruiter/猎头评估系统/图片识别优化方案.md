### **1. 图像预处理模块**

```python
class ImagePreprocessor:
    """智能图像预处理"""
    
    def preprocess(self, image_path):
        """
        多级预处理流程
        1. 质量评估
        2. 自适应处理
        3. 优化输出
        """
        steps = []
        
        # 步骤1: 读取并评估
        img = cv2.imread(image_path)
        quality_score = self.assess_quality(img)
        
        # 步骤2: 标准化大小
        if img.shape[1] > 2000:
            img = self.resize_to_width(img, 1500)
            steps.append("调整大小")
        
        # 步骤3: 灰度转换
        if len(img.shape) == 3:
            img = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
            steps.append("灰度转换")
        
        # 步骤4: 对比度增强（自适应）
        contrast_score = self.calculate_contrast(img)
        if contrast_score < 0.3:
            img = cv2.createCLAHE(clipLimit=3.0, tileGridSize=(8,8)).apply(img)
            steps.append("对比度增强")
        
        # 步骤5: 去噪
        img = cv2.medianBlur(img, 3)
        steps.append("去噪处理")
        
        # 步骤6: 锐化
        kernel = np.array([[-1,-1,-1], [-1,9,-1], [-1,-1,-1]])
        img = cv2.filter2D(img, -1, kernel)
        steps.append("锐化处理")
        
        # 步骤7: 二值化（自适应阈值）
        img = cv2.adaptiveThreshold(img, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, 
                                   cv2.THRESH_BINARY, 11, 2)
        steps.append("二值化")
        
        return img, steps
```

### **2. 智能OCR模块**

```python
class SmartOCR:
    """智能OCR识别"""
    
    def recognize(self, image, image_type="resume"):
        """
        智能识别，自动选择最佳参数
        """
        # 参数库
        param_library = {
            "resume": [
                {"psm": 6, "oem": 1, "lang": "chi_sim+eng"},
                {"psm": 3, "oem": 1, "lang": "chi_sim+eng"},
                {"psm": 4, "oem": 1, "lang": "chi_sim+eng"}
            ],
            "table": [
                {"psm": 4, "oem": 1, "lang": "chi_sim+eng"},
                {"psm": 11, "oem": 1, "lang": "chi_sim+eng"}
            ]
        }
        
        best_result = ""
        best_score = 0
        
        # 尝试多个参数组合
        for params in param_library.get(image_type, param_library["resume"]):
            try:
                # 临时保存图片
                temp_path = self.save_temp_image(image)
                
                # 构建命令
                cmd = self.build_tesseract_command(temp_path, params)
                
                # 执行OCR
                result = subprocess.run(cmd, capture_output=True, text=True, timeout=15)
                text = result.stdout.strip()
                
                # 评估结果质量
                score = self.evaluate_ocr_quality(text)
                
                if score > best_score:
                    best_score = score
                    best_result = text
                    
            except Exception as e:
                continue
        
        return best_result, best_score
```

### **3. 结构化信息提取模块**

```python
class ResumeParser:
    """简历结构化解析"""
    
    def parse(self, ocr_text):
        """
        从OCR文本中提取结构化信息
        """
        info = {
            "basic_info": {},
            "education": [],
            "experience": [],
            "investment": {},
            "skills": [],
            "expectations": {}
        }
        
        # 段落分割
        paragraphs = self.segment_paragraphs(ocr_text)
        
        # 段落分类
        classified = self.classify_paragraphs(paragraphs)
        
        # 信息提取
        for para_type, para_text in classified.items():
            if para_type == "personal":
                info["basic_info"] = self.extract_personal_info(para_text)
            elif para_type == "education":
                info["education"] = self.extract_education(para_text)
            elif para_type == "work":
                info["experience"] = self.extract_experience(para_text)
            elif para_type == "investment":
                info["investment"] = self.extract_investment_info(para_text)
        
        return info
```
